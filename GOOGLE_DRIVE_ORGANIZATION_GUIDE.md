## ğŸ¤– AUTOMATION SCRIPT

### **Google Apps Script: Auto-Organize by Visa Type**

Save this as a script in your Google Drive:

```javascript
/**
 * VISA EXHIBIT AUTO-ORGANIZER
 * Automatically categorizes files in a Drive folder based on visa type
 */

function autoOrganizeVisaExhibits() {
  // Get folder to organize
  var folderId = Browser.inputBox('Enter folder ID or URL:');
  var folder = getFolderFromIdOrUrl(folderId);

  // Get visa type
  var visaType = Browser.inputBox('Visa Type (O-1A, O-1B, P-1A, EB-1A):');

  // Create organized structure
  var structure = createFolderStructure(folder, visaType);

  // Categorize existing files
  categorizeFiles(folder, structure, visaType);

  Browser.msgBox('Organization complete! Check folder: ' + folder.getName());
}

function createFolderStructure(parentFolder, visaType) {
  var structure = {
    'Forms': parentFolder.createFolder('01-Forms'),
    'Personal': parentFolder.createFolder('02-Personal'),
    'Brief': parentFolder.createFolder('03-Brief'),
    'Contract': parentFolder.createFolder('04-Contract')
  };

  // Visa-specific folders
  if (visaType === 'O-1A') {
    structure['Criterion-A-Awards'] = parentFolder.createFolder('05-Criterion-A-Awards');
    structure['Criterion-C-Media'] = parentFolder.createFolder('06-Criterion-C-Media');
    structure['Criterion-E-Contributions'] = parentFolder.createFolder('07-Criterion-E-Contributions');
    structure['Criterion-G-Employment'] = parentFolder.createFolder('08-Criterion-G-Employment');
    structure['Criterion-H-Salary'] = parentFolder.createFolder('09-Criterion-H-Salary');
  } else if (visaType === 'P-1A') {
    structure['Itinerary'] = parentFolder.createFolder('05-Itinerary-REQUIRED');
    structure['Criterion-2-National-Team'] = parentFolder.createFolder('06-Criterion-2-National-Team');
    structure['Criterion-5-Experts'] = parentFolder.createFolder('07-Criterion-5-Expert-Statements');
    structure['Criterion-6-Rankings'] = parentFolder.createFolder('08-Criterion-6-Rankings');
    structure['Criterion-7-Awards'] = parentFolder.createFolder('09-Criterion-7-Awards');
  } else if (visaType === 'EB-1A') {
    structure['Criterion-3-Published'] = parentFolder.createFolder('04-Criterion-3-Published-Material');
    structure['Criterion-4-Judging'] = parentFolder.createFolder('05-Criterion-4-Judging');
    structure['Criterion-5-Contributions'] = parentFolder.createFolder('06-Criterion-5-Contributions');
    structure['Criterion-6-Articles'] = parentFolder.createFolder('07-Criterion-6-Scholarly-Articles');
    structure['Criterion-8-Employment'] = parentFolder.createFolder('08-Criterion-8-Employment');
  }

  structure['Supporting'] = parentFolder.createFolder('99-Supporting-Documents');

  return structure;
}

function categorizeFiles(folder, structure, visaType) {
  var files = folder.getFiles();

  while (files.hasNext()) {
    var file = files.next();
    var fileName = file.getName().toLowerCase();

    // Smart categorization
    if (fileName.includes('i-129') || fileName.includes('i-140') || fileName.includes('i-907')) {
      file.moveTo(structure['Forms']);
    } else if (fileName.includes('passport')) {
      file.moveTo(structure['Personal']);
    } else if (fileName.includes('brief') || fileName.includes('letter')) {
      file.moveTo(structure['Brief']);
    } else if (fileName.includes('contract') || fileName.includes('agreement')) {
      file.moveTo(structure['Contract']);
    } else if (fileName.includes('award') || fileName.includes('certificate')) {
      var targetFolder = visaType === 'P-1A' ? 'Criterion-7-Awards' : 'Criterion-A-Awards';
      if (structure[targetFolder]) file.moveTo(structure[targetFolder]);
    } else if (fileName.includes('article') || fileName.includes('media') || fileName.includes('espn')) {
      if (structure['Criterion-C-Media']) file.moveTo(structure['Criterion-C-Media']);
      if (structure['Criterion-3-Published']) file.moveTo(structure['Criterion-3-Published']);
    } else if (fileName.includes('expert') || fileName.includes('letter')) {
      if (structure['Criterion-E-Contributions']) file.moveTo(structure['Criterion-E-Contributions']);
      if (structure['Criterion-5-Contributions']) file.moveTo(structure['Criterion-5-Contributions']);
    } else if (fileName.includes('itinerary')) {
      if (structure['Itinerary']) file.moveTo(structure['Itinerary']);
    } else {
      // Move unclassified to Supporting
      file.moveTo(structure['Supporting']);
    }
  }
}

function getFolderFromIdOrUrl(input) {
  // Extract folder ID from URL or use as-is
  var folderId = input.match(/[-\w]{25,}/);
  return DriveApp.getFolderById(folderId ? folderId[0] : input);
}
```
